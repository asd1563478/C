#include <stdio.h>
 
int main()
{
  int c, n=5, fact = 1;
  for (c = 1; c <= n; c++)
    fact = fact * c;
 
  printf("Factorial of %d = %d\n", n, fact);
 
  return 0;
}

long int multiplyNumbers(int n);

int main()
{
   int n=5;
 
   printf("Factorial of %d = %ld", n, multiplyNumbers(n));

}
long int multiplyNumbers(int n)
{
   if (n >= 1)
       return n*multiplyNumbers(n-1);
   else
       return 1;
}
